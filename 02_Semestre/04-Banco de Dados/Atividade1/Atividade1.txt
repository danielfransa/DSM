Criação tabelas
CREATE DATABASE atividade1

CREATE TABLE cargos(
	idCargos INTEGER AUTO_INCREMENT,
	descricao VARCHAR (60),
	salario DECIMAL (10,2),
	PRIMARY KEY (idCargos)
)

CREATE TABLE funcionario(
	id INTEGER AUTO_INCREMENT,  
	idCargos INTEGER,
	nome VARCHAR (60),
	PRIMARY KEY (id),
	FOREIGN KEY (idCargos) REFERENCES cargos (idCargos) ON DELETE CASCADE ON UPDATE CASCADE   
)
