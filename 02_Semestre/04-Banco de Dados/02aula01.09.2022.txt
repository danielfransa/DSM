questão 8

questão 9
SELECT f.id, COUNT(p.idformapagto)
FROM pedido p, formapagto f
WHERE p.idformapagto = f.id
GROUP BY f.id

questão 10
SELECT p.id, p.idformapagto, f.descricao
FROM pedido p, formapagto f
WHERE f.id = p.idformapagto
AND p.datemissao BETWEEN '2019-01-01' AND '2019-06-30' 
ORDER BY p.id

SELECT p.id, p.idformapagto, f.descricao
FROM pedido p, formapagto f
WHERE f.id = p.idformapagto
AND p.datemissao >= '2019-01-01' 
AND p.datemissao <='2019-06-30' 
ORDER BY p.id
