Criar usuário no SQL

CREATE USER 'fatec'@'localhost' IDENTIFIED BY '1234'

Permissões

GRANT tipo-de-permissão ON nome-da-base-de-dados . nome-da-tabela TO 'nome-do-usuario'@'localhost'

Algumas permissões:

CREATE
DROP
DELETE
INSERT
SELECT
UPDATE
GRANT OPTION

código:
GRANT SELECT ON empresa2.cliente TO 'fatec'@'localhost';

GRANT SELECT, INSERT, UPDATE, DELETE ON empresa2.cliente TO 'fatec'@'localhost';

GRANT SELECT, INSERT, UPDATE, DELETE ON empresa2.cargo TO 'fatec'@'localhost';

GRANT SELECT ON empresa2.regiao TO 'fatec'@'localhost';

GRANT SELECT ON empresa2.representante TO 'fatec'@'localhost';

REVOKE DELETE ON empresa2.cargo FROM 'fatec'@'localhost';


backup banco
no mysql

banco todo
mysqldump.exe -u root -p empresa2 >empresa2_dump.sql

apenas uma tabela
mysqldump.exe -u root -p empresa2 empregado >empresa2_dump_emp.sql

restaurar o banco
mysql -u root -p empresa2 < empresa2_dump.sql

EXERCICIO

CREATE USER 'daniel'@'localhost' IDENTIFIED BY '1234'

GRANT SELECT, UPDATE ON empresa2.pedido TO 'daniel'@'localhost';

GRANT SELECT, UPDATE ON empresa2.item TO 'daniel'@'localhost';

GRANT SELECT, UPDATE ON empresa2.cliente TO 'daniel'@'localhost';

GRANT SELECT, INSERT ON empresa2.empregado TO 'daniel'@'localhost';

GRANT ALL PRIVILEGES ON empresa2.produto TO 'daniel'@'localhost';

REVOKE INSERT ON empresa2.empregado FROM 'daniel'@'localhost';

DROP USER 'daniel'@'localhost';

mysqldump.exe -u root -p empresa2 >empresa2_exer.sql

CREATE DATABASE empresa;
mysql -u root -p empresa < empresa2_exer.sql
